using namespace std;

int main(){

        int A, B, C, M;

            break;
        }

        string result;
        if(A > B){
            if(A > C){
                A * A == B * B + C * C ? result = "right" : result = "wrong";
            }else{
                C * C == A * A + B * B ? result = "right" : result = "wrong";
            }
        }else{
            if(B > C){
                B * B == A * A + C * C ? result = "right" : result = "wrong";
            }else{
                C * C == A * A + B * B ? result = "right" : result = "wrong";
            }
        }

        cout << result << "\n";
    }
    
    return 0;
}
